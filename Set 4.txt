#include<iostream>
using namespace std;
#include<algorithm>

int main()
{
	int m = 0, n = 0;
	cin >> m >> n;  

	while ((m !=0 && m>0) && (n !=0 && n>0) )
	{
		int sum = 0; 
		for (int i = min(m, n); i <= max(m, n); i++)
		{
			cout << i << " ";
			sum = sum + i;
		}
			cout<<"Sum=" << sum << endl;

			cin >> m >> n; 
	}
}
